   ```typescript
   export function TooltipIconButton({
       title,
       onClick,
       icon,
       placement = "top",
       color,
       size = "medium",
       disabled,
       sx,
   }: TooltipIconButtonProps): ReactElement {
       return (
           <Tooltip title={title} placement={placement}>
               <span>
                   <IconButton size={size} onClick={onClick} color={color} disabled={disabled} sx={sx}>
                       {icon}
                   </IconButton>
               </span>
           </Tooltip>
       );
   }
   ```

   Note the `<span>` wrapper around `<IconButton>` ã MUI Tooltip requires a DOM element that can receive focus/hover events, and disabled IconButtons don't forward events.
