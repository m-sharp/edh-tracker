  <files>lib/business/pod/functions.go</files>
  <action>
    Replace the Create constructor signature to accept a *lib.DBClient as a third argument:

    ```go
    func Create(podRepo repos.PodRepository, roleRepo repos.PlayerPodRoleRepository, client *lib.DBClient) CreateFunc {
    ```

    Replace the body with a single GORM transaction that:
    1. Inserts the pod row into table "pod" using an inline struct {Name string}, captures the auto-generated ID
    2. Inserts the player_pod row into table "player_pod" using PlayerPodModel{PodID: podID, PlayerID: creatorPlayerID}
    3. Calls roleRepo.SetRole(ctx, podID, creatorPlayerID, playerPodRole.RoleManager) — note: roleRepo methods accept context, not a tx; since SetRole only writes player_pod_role and is already covered by the outer transaction via MySQL autocommit semantics, call it via the outer repo after the transaction OR inline it against tx.

    IMPORTANT: roleRepo.SetRole uses its own internal db handle; it cannot participate in the
    GORM tx closure. Instead, do all three writes directly against `tx` inside the closure:

    ```go
    return func(ctx context.Context, name string, creatorPlayerID int) (int, error) {
        var podID int
        err := client.GormDb.WithContext(ctx).Transaction(func(tx *gorm.DB) error {
            // Step 1: insert pod
            type podRow struct {
                ID   int    `gorm:"primaryKey;autoIncrement"`
                Name string
            }
            pr := podRow{Name: name}
            if err := tx.Table("pod").Create(&pr).Error; err != nil {
                return fmt.Errorf("failed to insert pod in Create: %w", err)
            }
            podID = pr.ID

            // Step 2: insert player_pod
            type playerPodRow struct {
                PodID    int `gorm:"column:pod_id"`
                PlayerID int `gorm:"column:player_id"`
            }
            pp := playerPodRow{PodID: podID, PlayerID: creatorPlayerID}
            if err := tx.Table("player_pod").Create(&pp).Error; err != nil {
                return fmt.Errorf("failed to insert player_pod in Create: %w", err)
            }

            // Step 3: insert player_pod_role
            type podRoleRow struct {
                PodID    int    `gorm:"column:pod_id"`
                PlayerID int    `gorm:"column:player_id"`
                Role     string `gorm:"column:role"`
            }
            pr2 := podRoleRow{PodID: podID, PlayerID: creatorPlayerID, Role: playerPodRole.RoleManager}
            if err := tx.Table("player_pod_role").Create(&pr2).Error; err != nil {
                return fmt.Errorf("failed to insert player_pod_role in Create: %w", err)
            }

            return nil
        })
        if err != nil {
            return 0, fmt.Errorf("failed to create pod: %w", err)
        }
        return podID, nil
    }
    ```

    Add import for "gorm.io/gorm" in the functions.go import block. The lib package is already
    accessible as the import path "github.com/m-sharp/edh-tracker/lib" — add it if not present.
